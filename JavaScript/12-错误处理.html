JavaScript 错误处理

1. try...catch语句

基本语法

try {
    // 可能出错的代码
    let result = riskyOperation();
    console.log("结果:", result);
} catch (error) {
    // 处理错误
    console.log("发生错误:", error.message);
}

说明：try块中的代码如果出错，控制权会转移到catch块

捕获特定类型错误

try {
    let obj = null;
    console.log(obj.property); // 会抛出TypeError
} catch (error) {
    if (error instanceof TypeError) {
        console.log("类型错误:", error.message);
    } else if (error instanceof ReferenceError) {
        console.log("引用错误:", error.message);
    } else {
        console.log("其他错误:", error.message);
    }
}

说明：可以根据不同类型的错误进行不同的处理

2. finally子句

总是执行的代码

try {
    // 可能出错的代码
} catch (error) {
    // 错误处理
} finally {
    // 无论是否出错都会执行
    console.log("清理工作完成");
    // 比如关闭文件、释放资源等
}

说明：finally块中的代码无论是否出错都会执行

3. throw语句

手动抛出错误

function validateAge(age) {
    if (age < 0 || age > 150) {
        throw new Error("年龄必须在0-150之间");
    }
    return age;
}

try {
    validateAge(-5);
} catch (error) {
    console.log("验证失败:", error.message);
}

说明：使用throw关键字手动抛出异常

自定义错误类型

class ValidationError extends Error {
    constructor(message) {
        super(message);
        this.name = "ValidationError";
    }
}

function checkEmail(email) {
    if (!email.includes("@")) {
        throw new ValidationError("邮箱格式不正确");
    }
}

说明：可以创建自定义的错误类来更好地组织错误处理

4. Promise错误处理

Promise.catch()

fetchData()
    .then(result => processResult(result))
    .then(finalResult => console.log(finalResult))
    .catch(error => {
        console.log("Promise错误:", error);
    });

说明：Promise链式的错误处理方式

async/await中的错误处理

async function getData() {
    try {
        let data = await fetchData();
        return processData(data);
    } catch (error) {
        console.log("获取数据失败:", error);
        throw error; // 重新抛出或返回默认值
    }
}

说明：在async函数中使用try-catch处理异步错误

5. 实际应用示例

表单验证

function validateForm(formData) {
    if (!formData.username) {
        throw new Error("用户名不能为空");
    }
    if (formData.password.length < 6) {
        throw new Error("密码长度至少6位");
    }
    return true;
}

说明：在表单提交时进行验证并抛出相应的错误

API调用错误处理

async function apiCall(url) {
    try {
        const response = await fetch(url);
        if (!response.ok) {
            throw new Error(`HTTP错误! 状态: ${response.status}`);
        }
        return await response.json();
    } catch (error) {
        console.error("API调用失败:", error);
        return null;
    }
}

说明：网络请求时的完整错误处理流程
